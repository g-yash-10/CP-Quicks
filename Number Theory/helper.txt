static int factorial(int n){
        int res = 1, i;
        for (i = 2; i <= n; i++)
            res *= i;
        return res;
}

//O(log(min(a,b)))
static int gcd(int a, int b){
      if (b == 0)
        return a;
      return gcd(b, a % b);
}


// a x b = LCM(a,b) * GCD(a,b)
static int lcm(int a, int b){
        return (a / gcd(a, b)) * b;
}
